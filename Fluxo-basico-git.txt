git init --> inicia um repositório local  

git remote add origin "link do repositório do gitHub"

git add . --> adicionad todos os arquivos para staged (ponto quer dizer que adiciona todos os arquivos)

git commit -m "mensagem do commit" 

git push origin main 

git status --> ver os arquivos que estão untracked, staged, etc.

*agora estamos na branch main*

git branch -M main --> Renomeia a branch principal 

git checkout -b "nome da branch" --> trocar para outra branch

git branch --> mostra as branch existentes
